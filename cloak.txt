function cloakPrompt() {
  let text;
  let site = prompt("Website to open cloaked:", "");
  if (site == null || site == "") {
    void 0;
  } else {
    var win = window.open()
    var url = site
    var iframe = win.document.createElement('iframe')
    iframe.style.width = "100%";
    iframe.style.height = "100%";
    iframe.style.border = "none";
    iframe.src = url
    win.document.body.appendChild(iframe)
  }
}
cloakPrompt()
